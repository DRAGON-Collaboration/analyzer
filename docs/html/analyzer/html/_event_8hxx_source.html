134  /// Bank finding routine (templated)
135  template <typename T>
136  T* GetBankPointer(const Bank_t name, int* length, bool reportMissing = false, bool checkType = false) const
137  {
138  /*!
139  * \param [in] name Name of the bank to search for
140  * \param [out] Length of the returned bank
141  * \param [in] reportMissing True means a warning message is printed if the bank is absent
142  * \param [in] checkType Specifies whether or not to check that the template parameter
143  * matches the TID of the bank. If this parameter is set to true and the types do
144  * not match, the error is fatal.
145  * \returns Pointer to the beginning of the bank
146  */
147  void *pbk;
148  int type;
149  int bkfound = FindBank(name, length, &type, &pbk);
150 
151  if(!bkfound && reportMissing) {
152  dragon::utils::Warning("midas::Event::GetBankPointer<T>", __FILE__, __LINE__)
153  << "Couldn't find the MIDAS bank \"" << name << "\". Skipping...\n";
154  }
155  if (bkfound && checkType) {
156  switch (type) {
157  case 1: assert (typeid(T) == typeid(unsigned char)); break; // TID_BYTE 1
158  case 2: assert (typeid(T) == typeid(char)); break; // TID_SBYTE 2
159  case 3: assert (typeid(T) == typeid(unsigned char)); break; // TID_CHAR 3
160  case 4: assert (typeid(T) == typeid(uint16_t)); break; // TID_WORD 4
161  case 5: assert (typeid(T) == typeid(int16_t)); break; // TID_SHORT 5
162  case 6: assert (typeid(T) == typeid(uint32_t)); break; // TID_DWORD 6
163  case 7: assert (typeid(T) == typeid(int32_t)); break; // TID_INT 7
164  case 8: assert (typeid(T) == typeid(bool)); break; // TID_BOOL 8
165  case 9: assert (typeid(T) == typeid(float)); break; // TID_FLOAT 9
166  case 10: assert (typeid(T) == typeid(double)); break; // TID_DOUBLE 10
167  default:
168  fprintf(stderr, "Unknown type id: %i\n", type);
169  assert(false); break;
170  }
171  }
172  return bkfound ? reinterpret_cast<T*>(pbk) : 0;
173  }
