18 /// Class to manage coincidence/singles ID.
19 /*!
20  * The basic idea is to buffer events in a queue for long enough to ensure that any possible
21  * coincidence events have been received from the data source. Then, check if the early event
22  * has any coincidence matches, and, if so, to process the events together. Then the earliest
23  * event is processed as a singles event and removed from the queue.
24  *
25  * The interface consists of Push() and Pop() functions to insert and remove events from the
26  * queue, Size() to get the queue length, and Flush() to explicitly empty the queue. To
27  * set how coincidence and singles events should be handled, the class provides the private
28  * virtual functions HandleCoinc() and HandleSingle(). In the base class, these simply
29  * print some information about the arguments to stdout.
30  */
31 class Queue {
32 public:
33  /// Internal container type used to store events.
34  /*!
35  * Performance testing has indicated that a `std::multiset` is the optimal container
36  * for our needs:
37  * - Frequent insertion into the container
38  * - Frequent removal of the oldest element in the container
39  * - Quick identification of (possible multiple) coincidences by searching on the trigger time
40  *
41  * Others containers tested included a sorted `std::deque` or `std::vector`; `boost::container::flat_multiset`
42  * (essentially a sorted `std::vector`); and `boost::unordered_multiset`. The sorted `std::deque` performed
43  * similarly to `std::multiset` for objects with cheap copy operations (e.g. a simple 'int' or 'double'),
44  * but for objects with non-trivial copy constructors the extra copy operations required to maintain
45  * sorting inhibit performance significantly (note that `std::vector` had bad performance in all cases
46  * due to the expense of removing items at the beginning of the vector).
47  *
48  * The boost hash implementation (`boost::unordered_multiset`)
49  * showed slightly better performance for coincidence matching than
50  * the `std::multiset`; however, the the increase was not significant, and it is not likely that it's
51  * worth the extra code required to keep track of the earliest entry and the the reliance
52  * on "non-standard" third-party code for a hash implementation.
53  */

139 /// Queue that is a member of another class which handles popped events
140 /*!
141  * The intended use of this class is for when the queue exists as a data member of
142  * some other class (the "owner"). The owning class will perform the work of handling
143  * singles or coincidence events popped from the queue (see below).
144  * \tparam T The type of class owning the queue; must have the following member functions:
145  *
146  * - <tt> void Process(const midas::Event&); </tt>
147  * - <tt> void Process(const midas::Event&, cont midas::Event&); </tt>
148  * - <tt> void Process(tstamp::Diagnostics*); </tt>
149  *
150  * to handle singles and coincidence events, respectively.
151  */
152 template <class T>
153 class OwnedQueue: public Queue {
154 private:
155  /// Reference to the class "owning" the queue
156  T& fOwner;
157 
158 public:
159  /// Calls base constructor with maxDelta argument; sets fOwner
160  OwnedQueue(double maxDelta, T* owner):
161  tstamp::Queue(maxDelta), fOwner(*owner) { }
162 
163  /// Empty
164  ~OwnedQueue() { }
165 
166 private:
167  /// Calls <tt> fOwner.Process(const midas::Event&); </tt>
168  void HandleSingle(const midas::Event& event) const
169  { fOwner.Process(event); }
170 
171  /// Calls <tt> fOwner.Process(const midas::Event&, const midas::Event&); </tt>
172  void HandleCoinc(const midas::Event& event1, const midas::Event& event2) const
173  { fOwner.Process(event1, event2); }
174

176  void HandleDiagnostics(tstamp::Diagnostics* diagnostics) const
