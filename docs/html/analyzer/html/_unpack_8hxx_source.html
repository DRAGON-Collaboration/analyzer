184 inline void dragon::Unpacker::SetSinglesMode(int qFlush)
185 {
186  /// Any incoming events after this call are processed as singles only.
187  /// If there are events in the queue, they are first flushed.
188  /// \param qFlush Maximum flush time for events in the queue. Specifying a
189  /// 0 skips flushing entirely and throws away any events in the queue. Negative
190  /// numbers cause the entire queue to be flushed.
191  if(qFlush) FlushQueue(qFlush);
192  fQueue.reset(0);
193 }
194 
195 inline void dragon::Unpacker::SetCoincMode()
196 {
197  /// If in singles mode when called, it switches over to coincidence
198  /// mode with default coincidence window and queue time. To change
199  /// these values, use SetCoincWindow() and SetQueueTime().
200  ///
201  /// If we are already in coincidence mode, this has no effect.
202  if(!IsSinglesMode())
203  fQueue.reset(new tstamp::OwnedQueue<Unpacker>(kQueueTimeDefault*1e6, this));
204 }
