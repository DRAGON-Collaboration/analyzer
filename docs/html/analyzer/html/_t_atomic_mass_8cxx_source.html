140 bool TAtomicMassTable::ParseLine(const std::string& line, Nucleus_t* nuc, MassExcess_t* me) const
141 {
142  size_t lengths[11] = { 1, 3, 5, 5, 5, 1, 3, 4, 1, 13, 11 };
143  std::string fields[11];
144 
145  size_t sum = 0;
146  for(int i=0; i< 11; ++i) {
147  fields[i] = line.substr(sum, lengths[i]);
148  sum += lengths[i];
149  // std::cout << fields[i] << " | "; if(i==10) std::cout << "\n";
150  }
151 
152  // N collumn 2
153  nuc->fN = atoi(fields[2].c_str());
154  // Z collumn 3
155  nuc->fZ = atoi(fields[3].c_str());
156  // A collumn 4
157  nuc->fA = atoi(fields[4].c_str());
158  if(nuc->fA != nuc->fZ + nuc->fN) {
159  std::cerr << "Error: A != N+Z : A, N, Z = "
160  << nuc->fA << ", " << nuc->fZ << ", " << nuc->fN << "\n";
161  return false;
162  }
163 
164  // symbol collumn 6
165  {
166  if(fields[6].size() != 3) {
167  std::cerr << "Error: fields[6].size != 3, == " << fields[6].size() << "\n";
168  return false;
169  }
170  int pos = 0;
171  for(size_t i=0; i< fields[6].size(); ++i) {
172  if(fields[6][i] != ' ')
173  nuc->fSymbol[pos++] = fields[6][i];
174  }
175  nuc->fSymbol[pos] = '\0';
176  }
177 
178  // mass excess collumn 9
179  {
180  me->fExtrapolated = false;
181  if(fields[9].find("#") < fields[9].size()) {
182  me->fExtrapolated = true;
183  fields[9] = fields[9].substr(0, fields[9].find("#"));
184  }
185 
186  me->fValue = atof(fields[9].c_str());
187  }
188 
189  // mass excess error collumn 10
190  if(me->fExtrapolated)
191  fields[10] = fields[10].substr(0, fields[10].find("#"));
192  me->fError = atof(fields[10].c_str());
193 
194  return true;
195 }
