184 void dragon::Bgo::calculate()
185 {
186  /*!
187  * Does the following:
188  */
189  /// - Pedestal subtract, zero suppress and calibrate energy values
190  dutils::pedestal_subtract(ecal, MAX_CHANNELS, variables.adc);
191  dutils::zero_suppress1(ecal, MAX_CHANNELS, 10.);
192  dutils::linear_calibrate(ecal, MAX_CHANNELS, variables.adc);
193 
194  /// - Calibrate time values
195  dutils::linear_calibrate(tcal, MAX_CHANNELS, variables.tdc);
196 
197  /// - Calculate descending-order energy indices and map into \c esort[]
198  int isort[MAX_CHANNELS];
199  dutils::index_sort(ecal, ecal+MAX_CHANNELS, isort, dutils::greater_and_valid<double>());
200  dutils::channel_map_from_array(esort, MAX_CHANNELS, isort, ecal);
201 
202  /// - If we have at least one good hit, calculate sum, x0, y0, z0, and t0
203  if(dutils::is_valid(esort[0])) {
204  hit0 = isort[0];
205  sum = dutils::calculate_sum(ecal, ecal + MAX_CHANNELS);
206  x0 = variables.pos.x[ isort[0] ];
207  y0 = variables.pos.y[ isort[0] ];
208  z0 = variables.pos.z[ isort[0] ];
209  t0 = tcal[ isort[0] ];
210  }

518 void dragon::Mcp::calculate()
519 {
520  /*!
521  * Calibrates anode, tcal, and tac values; calculates x- and
522  * y-positions.
523  *
524  * \note Position calculation algorithm taken from the MSc thesis of
525  * Michael Lamey, Simon Fraser University, 2001, p. 25; available online at
526  * <a href="http://dragon.triumf.ca/docs/Lamey_thesis.pdf">
527  * dragon.triumf.ca/docs/Lamey_thesis.pdf</a>
528  */
529  dutils::linear_calibrate(anode, MAX_CHANNELS, variables.adc);
530  dutils::linear_calibrate(tcal, NUM_DETECTORS, variables.tdc);
531  dutils::linear_calibrate(tac, variables.tac_adc);
532 
533  dutils::calculate_sum(anode, anode + MAX_CHANNELS);
534 
535  // Position calculation if we have all valid anode signals
536  if(dutils::is_valid_all(anode, MAX_CHANNELS)) {
537  const double Lhalf = 25.; // half the length of a single side of the MCP (50/2 [mm])
538  double sum = 0;
539  for(int i=0; i< MAX_CHANNELS; ++i) sum += anode[i];
540  if(sum != 0) {
541  x = Lhalf * ( (anode[1] + anode[2]) - (anode[0] + anode[3]) ) / sum;
542  y = Lhalf * ( (anode[0] + anode[1]) - (anode[2] + anode[3]) ) / sum;
543  }
544  }
545 }
