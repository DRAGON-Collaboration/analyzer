225  /// Parse command line arguments
226  int process_args(int argc, char** argv, Options_t* options)
227  {
228  strvector_t args(argv+1, argv+argc);
229  strvector_t::iterator iarg = args.begin();
230 
231  //
232  // Look for input file
233  for(; iarg != args.end(); ++iarg) {
234  if(iarg->substr(0, 2) == "--")
235  continue;
236  if((iarg-1 >= args.begin()) && *(iarg-1) == "--quiet")
237  continue;
238  options->fIn = *iarg;
239  break;
240  }
241 
242  //
243  // Check other arguments
244  for(iarg = args.begin(); iarg != args.end(); ++iarg) {
245  if (*iarg == options->fIn) { // Input file, already set
246  continue;
247  }
248  else if (*iarg == "--help") { // Help message
249  return help();
250  }
251  else if (*iarg == "-o") { // Output file
252  if (++iarg == args.end()) return usage("output file not specified");
253  options->fOut = *iarg;
254  }
255  else if (*iarg == "-v") { // Variables file
256  if (++iarg == args.end()) return usage("variables file not specified");
257  options->fOdb = *iarg;
258  }
259  else if (*iarg == "-histos") { // Histograms file
260 #ifndef USE_ROOTBEER
261  return usage("histogram unpacking only available if compiled with USE_ROOTBEER=YES");
262 #endif
263  if (++iarg == args.end()) return usage("histogram xml file not specified");
264  options->fHistos = *iarg;
265  }
266  else if (*iarg == "--singles") { // Singles mode
267  options->fSingles = true;
268  }
269  else if (*iarg == "--sonik") { // SONIK mode
270  options->fSonik = true;
271  }
272  else if (*iarg == "--overwrite") { // Overwrite flag
273  options->fOverwrite = true;
274  }
275  else if (*iarg == "--quiet") { // Quiet flag
276  if (++iarg == args.end()) return usage("quietness level not specified");
277  TString qstr = iarg->c_str();
278  if (qstr.IsDigit() == false) {
279  TString error ("Quietness level \'");
280  error += qstr; error += "\' is not an integer";
281  return usage(error.Data());
282  }
283  m2r::SetQuietLevel(qstr.Atoi());
284  }
285  else { // Unknown flag
286  TString what = "unknown flag \'";
287  what += *iarg; what += "\'";
288  usage(what.Data());
289  return 1;
290  }
291  }
292 
293  if (options->fIn.empty()) // Didn't find input file
294  return usage("no input file specified");
295 
296  return 0;
297  }

731  //
732  // Rootbeer associates histograms with a data address using
733  // classes derived from rb::Event, which contain a wrapper
734  // to classes holding user data. However, here we don't want
735  // to use all of this, so we can fake it by just copying
736  // the data from `head`, `tail`, etc that are defined in main_().
737  // To make this generic we can use some template and inheritance
738  // tricks. First define an abstract with a `SetData()` function
739  // to copy data.
740  class AEvent: public rb::Event {
741  public:
742  virtual void SetData(const void* addr) = 0;
743  };
744 
745  //
746  // Then define a template derived class that fills in all of the
747  // implementations generically.
748  template <class T, const char* STR, bool B>
749  class EventTemplate: public AEvent {
750  protected:
751  rb::data::Wrapper<T> fWrapper; // data wrapper - what we copy to
752  public:
753  //
754  // Initialize data wrapper with template arguments
755  EventTemplate(): fWrapper(STR, this, B, "") { }
756  //
757  // Copy data at an address to the data wrapper
758  void SetData(const void* addr) { *fWrapper = *reinterpret_cast<const T*>(addr); }
759  private:
760  //
761  // Required pure virtual functions - implement with nothing
762  Bool_t DoProcess(const void*, Int_t) { return true; }
763  void HandleBadEvent() { }
764  };
