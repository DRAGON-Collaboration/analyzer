107 void rbdragon::MidasBuffer::RunStartTransition(Int_t runnum)
108 {
109  ///
110  /// Does the following:
111 
112  /// - Reset scalers
113  rb::Event::Instance<rbdragon::HeadScaler>()->Reset();
114  rb::Event::Instance<rbdragon::TailScaler>()->Reset();
115  rb::Event::Instance<rbdragon::AuxScaler>()->Reset();
116  rb::Event::Instance<rbdragon::RunParameters>()->Reset();
117  rb::Event::Instance<rbdragon::TStampDiagnostics>()->Reset();
118 
119  /// - Read variables from the ODB if online and if told to (SetAutoZeroOdb)
120  if (fType == rb::MidasBuffer::ONLINE && GetAutoZeroOdb() == true) {
121  dragon::utils::Info("rbdragon::MidasBuffer")
122  << "Syncing variable values with the online database.";
123  midas::Database db("online");
124  ReadVariables(&db);
125 
126  // Also set auto zero level from "/dragon/rootbeer/AutoZero"
127  {
128  Int_t autoZeroLevel = 0;
129  Bool_t success = db.ReadValue("/dragon/rootbeer/AutoZero", autoZeroLevel);
130  if(success) {
131  SetAutoZero(autoZeroLevel);
132  }
133  else {
134  dragon::utils::Warning("RunStartTransition")
135  << "Unable to read auto zero level from MIDAS database \"/dragon/rootbeer/AutoZero\", "
136  << "defaulting to level 1.";
137  SetAutoZero(1);
138  }
139  }
140  }
141 
142  /// - Zero all histograms if online and enabled:
143  if (fType == rb::MidasBuffer::ONLINE) {
144  switch (GetAutoZero()) {
145  case 0: /// - Level 0 - do nothing
146  break;
147  case 1: /// - Level 1 - clear scalers only
148  {
149  Int_t codes[] = { DRAGON_HEAD_SCALER, DRAGON_TAIL_SCALER, DRAGON_AUX_SCALER };
150  Int_t ncodes = sizeof(codes) / sizeof(Int_t);
151  for(Int_t i=0; i< ncodes; ++i) {
152  rb::Event* evt = rb::Rint::gApp()->GetEvent (codes[i]);
153  if(!evt) continue;
154  rb::hist::Manager* manager = evt->GetHistManager();
155  if(manager) manager->ClearAll();
156  }
157  }
158  break;
159  case 2: /// - Level 2 - clear everything
160  rb::hist::ClearAll();
161  break;
162  default:
163  break;
164  }
165  }
166 
167  /// - Look at ODB and set canvas to web saving if enabled
168  if (fType == rb::MidasBuffer::ONLINE) {
169  midas::Database db("online");
170 
171  std::string webFile = "";
172  Bool_t saveCanvas = true, success = kTRUE;
173  if(success) saveCanvas = midas::Odb::ReadBool("/dragon/rootbeer/canvas/SaveToWeb");
174  if(success) success = db.ReadValue("/dragon/rootbeer/canvas/WebFile", webFile);
175 
176  if(success && !webFile.empty()) {
177  rb::canvas::SetWebFile(webFile.c_str());
178  } else {
179  if(!success) {
180  dragon::utils::Warning("RunStartTransition")
181  << "Couldn't read web canvas settings from ODB";
182  } else {
183  dragon::utils::Warning("RunStartTransition")
184  << "Empty \"/dragon/rootbeer/WebFile\" in ODB";
185  }
186  }
187  }

189  /// - Call parent class implementation (prints a message)
190  rb::MidasBuffer::RunStartTransition(runnum);
191 }

211 void rbdragon::MidasBuffer::RunStopTransition(Int_t runnum)
212 {
213  ///
214  /// Does the following:
215 
216  /// - Flush timestamp queue (max 15 seconds for online)
217  Int_t flush_time = fType == ONLINE ? 15 : -1;
218  time_t t_begin = time(0);
219  size_t qsize;
220  while (1) {
221  if (flush_time >= 0 &&
222  (difftime(time(0), t_begin) > flush_time))
223  break;
224 
225  fUnpacker.ClearUnpackedCodes();
226  qsize = fUnpacker.FlushQueueIterative(); // Fills classes implicitly
227  if (qsize == 0) break;
228 
229  std::vector<Int_t> v = fUnpacker.GetUnpackedCodes();
230  process_events(v);
231  }
232  if (qsize) {
233  fUnpacker.GetQueue()->FlushTimeoutMessage(flush_time);
234  fUnpacker.GetQueue()->Clear();
235  }
236 
237  /// - Print delayed error messages
238  dragon::utils::gDelayedMessageFactory.Flush();
239 
240  /// - Call parent class implementation (prints a message)
241  rb::MidasBuffer::RunStopTransition(runnum);
242 }

255 Bool_t rbdragon::MidasBuffer::UnpackEvent(void* header, char* data)
256 {
257  ///
258  /// Does the following:
259 
260  midas::Event::Header* phead = reinterpret_cast<midas::Event::Header*>(header);
261 
262  /// - If BOR event (ODB dump), set variables from there
263  if (phead->fEventId == MIDAS_BOR) {
264  midas::Database db(data, phead->fDataSize);
265  ReadVariables(&db);
266  }
267 
268  /// - For all other events, delegate to rb::Unpacker, then call the Process() function
269  std::vector<Int_t> v = fUnpacker.UnpackMidasEvent(phead, data);
270  process_events(v);
271 
272  return kTRUE;
273 }
