20 // Helper function to calculate full timestamp values
21 inline uint64_t read_timestamp (uint32_t tscl, uint32_t tsch)
22 {
23  // tscl: lower bits 29..0 bits, tsch: upper bits 35..28
24  uint64_t tscfull = tscl & 0x1fffffff; // bits 28..0
25 
26  // check for match in overlap bit 29
27  int bit29h = (tsch>>1) & 0x1, bit29l = (tscl>>29) & 0x1;
28  if (bit29h != bit29l) { // adjust upper bits to account for difference
29  if (bit29l < bit29h) ++tsch;
30  else --tsch;
31  }
32  // append bits > 29
33  tscfull |= ((uint64_t)(tsch>>1) << (uint64_t)29);
34  return tscfull;
35 }

139 void midas::Event::Init(const char* tsbank, const void* header, const void* addr, int size)
140 {
141  memcpy(GetEventHeader(), header, sizeof(midas::Event::Header));
142  memcpy(GetData(), addr, GetDataSize());
143  SetBankList();
144 
145  if (tsbank != 0) {
146  int tsclength;
147  uint32_t* ptsc = GetBankPointer<uint32_t> (tsbank, &tsclength, true, true);
148  if (!ptsc) throw(std::invalid_argument(tsbank));
149 
150  // Read: firmware revision, write timestamp, routing, sync number
151  uint32_t version = *ptsc++;
152  uint32_t bkts = *ptsc++;
153  uint32_t route = *ptsc++;
154 
155  // Suppress compiler warning about unused values
156  if (version && bkts && route)
157  ;
158 
159  // Check version
160  uint32_t versions[] =
161  { 0x01130215, 0x01121212, 0x01120925, 0x01120809, 0x01120810, 0x01120910 };
162 
163  bool okVersion = false;
164  for(uint32_t v=0; v< sizeof(versions)/sizeof(uint32_t); ++v) {
165  if(version == versions[v]) { okVersion = true; break; }
166  }
167  if(okVersion == false) {
168  dragon::utils::Warning("midas::Event::Init") <<
169  "Unknown TSC version 0x" << std::hex << version << std::dec << " (id, serial #: " << GetEventId() <<
170  ", " << GetSerialNumber() << ")" << DRAGON_ERR_FILE_LINE;
171  }
172 
173  // Get TSC4 info
174  uint32_t ctrl = *ptsc++, roll = *ptsc++;
175  uint32_t nch = ctrl & READ14; // number of channels
176  bool overflow = (ctrl>>15) & READ1; // overflow flag
177  uint32_t tsch = (ctrl>>16) & READ8; // upper tsc bits 35..28
178 
179  if (overflow && false) {
180  dragon::utils::Warning("midas::Event::Init") <<
181  "IO32 TSC in overflow condition. Event Serial #, Id: " << GetSerialNumber() << ", " << GetEventId() << "\n";
182  }
183 
184  for(uint32_t i=0; i< nch; ++i) {
185  uint32_t tscl = *ptsc++, ch = (tscl>>30) & READ2;
186  assert(ch< MAX_FIFO);
187 
188  uint64_t tscfull = read_timestamp(tscl, tsch | (roll<<8));
189  fFifo[ch].push_back(tscfull);
190 
191  if(ch == TRIGGER_CHANNEL && fClock == std::numeric_limits<uint64_t>::max()) {
192  fClock = tscfull;
193  fTriggerTime = fClock / DRAGON_TSC_FREQ;
194  }
195  }
196  } // if tsbank != 0
197 }
