64 double dragon::LinearFitter::Chi2err(double M, double B, double x, double y, double exl, double exh, double eyl, double eyh)
65 {
66  double fx = M*x + B;
67  double ey = fx < y ? eyl : eyh;
68 
69  double num = (y-fx)*(y-fx);
70  double den = ey*ey + pow(0.5*M*(exl+exh), 2);
71  if(den == 0) // no errors
72  return num;
73  return num / den;
74 }
