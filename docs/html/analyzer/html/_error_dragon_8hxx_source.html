258 /*!

270  { if(message) fMessage = message; }

274  /// Returns message period, i.e. how many messages in between prints

276  /// Set message period

341 class DelayedMessageFactory {
342 public:
343  /// Empty
344  DelayedMessageFactory() { }
345  /// Free memory allocated to all registered printers
346  ~DelayedMessageFactory();
347  /// Register a new delayed message printer
348  template <class T, class U>
349  ADelayedMessagePrinter* Register(U base, int code, const char* location, int32_t period,
350  const char* file = "", int line = -1, const char* message = 0)
351  {
352  ///
353  /// \param base Base part of the key, often the address of the class associated w/ the message
354  /// \param code Offset part of the key, the final access key is `base + code`
355  /// \param location Where the message is originating from
356  /// \param file File from where the message originates
357  /// \param line Line from where the message originates
358  /// \param period The message period
359  /// \param message The message to print
360  /// \returns Pointer to the newly created message printer, or NULL if one already exists with key `base + code`
361 
362  int64_t key = (int64_t)base + code;
363  ADelayedMessagePrinter* err = new DelayedMessagePrinter<T>(location, period, file, line, message);
364  if(fPrinters.insert(std::make_pair(key, err)).second) return err;
365  else delete err;
366  Error("DelayedMessageFactory::Register")
367  << "Couldn't register delayed message printer for location \"" << location << "\", period " << period
368  << ", message \"" << message << "\": Key " << key << " already in use";
369  return 0;
370  }

373  template <class T>
374  ADelayedMessagePrinter* Get(T base, int code)
375  {

378  /// \param code Offset part of the key

380
