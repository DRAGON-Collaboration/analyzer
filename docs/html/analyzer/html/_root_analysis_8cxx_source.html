414 Int_t dragon::TTreeFilter::Run()
415 {
416  if(IsZombie()) {
417  utils::Error("TTreeFilter::Run", __FILE__, __LINE__) << "Zombie output directory.";
418  return -1;
419  }
420  if(fInputs.empty()) {
421  utils::Error("TTreeFilter::Run", __FILE__, __LINE__) << "No inputs to filter.";
422  return -1;
423  }
424  std::vector<std::pair<TThread*, ThreadArgs_t*> > threads;
425  threads.reserve(fInputs.size());
426  for(Map_t::iterator i = fInputs.begin(); i != fInputs.end(); ++i) {
427  if(CheckCondition(i->first) == kFALSE) {
428  utils::Warning("TTreeFilter::Run", __FILE__, __LINE__)
429  << "Invalid filter condition: \"" << GetFilterCondition(i->first)
430  << "\" for TTree at " << i->first << ", skipping.";
431  continue;
432  }
433  // Construct arguments
434  ThreadArgs_t *args = new ThreadArgs_t();
435  args->fIn = i->first;
436  args->fOut = &(i->second.fTree);
437  args->fOutDir = GetOutDir();
438  args->fCondition = i->second.fCondition;
439  // Construct thread and add to list
440  TThread *thread = GetThreaded() ? new TThread(run_thread, args) : 0;
441  threads.push_back(std::make_pair(thread, args));
442  }
443  // Initial message
444  std::cout
445  << "Running the following filters:\n"
446  << "\t<tree name>, <num events>, <filter condition>\n";
447  for(size_t i=0; i< threads.size(); ++i)
448  std::cout << "\t" << threads[i].second->fIn->GetName() << ", "

464  }
465  }

684 ////////////////////////////////////////////////////////////////////////////////
685 /// Calculate average current for a given cup reading and cup
686 /// \param run Run number from which to get cup readings; this will look at the
687 /// readings just before starting the specified run number
688 /// \param cup Specifies the farady cup from which to read, in order of
689 /// upstream -> downstream:
690 /// - 0 : FC4
691 /// - 1 : FC1
692 /// - 2 : FCCH
693 /// - 3 : FCM
694 /// - 4 : FCF
695 /// \param iteration If more than one cup reading was taken, specifices which
696 /// reading to take the average of (0 == first iteration).
697 /// \param skipBegin Length of time in seconds to skip at the beginning of the cup
698 /// readings
699 /// \param skipEnd Length of time in seconds to skip at the end of the cup readings
700 /// \returns Average current value for the parameters specified. This is returned
701 /// as a class that holds both the nominal average value and the 1-sigma
702 /// error. The 1-sigma error is calculated as the standard deviation of
703 /// each individual cup reading over the specified duration. Note that to
704 /// see the mean & error as part of a ROOT session, you can do
705 /// \code
706 /// rossumData->AverageCurrent(1000, 0, 0)->Print();
707 /// \endcode
708 UDouble_t dragon::RossumData::AverageCurrent(Int_t run, Int_t cup, Int_t iteration,
709  Double_t skipBegin, Double_t skipEnd)
710 {
711  TTree* tree = GetTree(run);
712  if(!tree) {
713  utils::Error("RossumData::AverageCurrent", __FILE__, __LINE__)
714  << "Invalid run " << run;
715  return UDouble_t(0);
716  }
717  Double_t time_, current_;
718  Int_t cup_, iteration_, success;
719  success = tree->SetBranchAddress("current", &current_);
720  success = tree->SetBranchAddress("time", &time_);
721  success = tree->SetBranchAddress("cup", &cup_);
722  success = tree->SetBranchAddress("iteration", &iteration_);
723  if(success) {
724  utils::Error("RossumData::AverageCurrent", __FILE__, __LINE__)
725  << "Failure to set branch addresses";
726  return UDouble_t(0);
727  }
728  AutoResetBranchAddresses Rst_(tree);
729  Double_t t0 = -1;
730  std::vector<Double_t> current, time;
731  for(Long64_t entry = 0; entry< tree->GetEntries(); ++entry) {
732  tree->GetEntry(entry);
733  if(cup_ != cup || iteration_ != iteration)
734  continue; // failed cut
735  assert(time_ >= 0);
736  if(t0 < 0) t0 = time_;
737  if(time_ - t0 > skipBegin) {
738  if(!time.empty() && !(time_ >= time.back())) {
739  utils::Error("RossumData::AverageCurrent", __FILE__, __LINE__)
740  << "Non sequential current readings: " << "this, last = "
741  << time_ << ", " << time.back() << "\n";
742  return UDouble_t(0, 0);
743  }
744  current.push_back(current_);
745  time.push_back(time_);
746  }
747  }
748  // Find last time
749  const std::vector<Double_t>::iterator::difference_type diffLast =
750  std::lower_bound(time.begin(), time.end(), time.back() - skipEnd,
751  std::less_equal<Double_t>()) - time.begin();
752  const std::vector<Double_t>::iterator lastCurrent =
753  current.begin() + diffLast;
754  Double_t avg = utils::calculate_mean(current.begin(), lastCurrent);
755  Double_t stddev = utils::calculate_stddev(current.begin(), lastCurrent, avg);
756  return UDouble_t(avg, stddev);
757 }

1021 /// Calculate R and total beam particles
1022 void dragon::BeamNorm::CalculateNorm(Int_t run, Int_t chargeState)
1023 {

1028  }

1600 /// \param [in] db Pointer to the "odbstop" database for the current run
1601 /// \param [in] which "head", "tail", or "coinc"
1602 /// \param [out] start Run start time (relative to TSC reset)
1603 /// \param [out] stop Run stop time (relative to TSC reset)
1604 /// \returns Total run time in seconds. Note that in the case of
1605 /// "coinc", the runtime is calculated as
1606 /// < smallest stop time > - < largest start time >, i.e. it is the
1607 /// total time during which _both_ acquisitions are taking triggers.
1608 ///
1609 
1610 // TSC stop is stored in the ODB with rollover, so first figure
1611 // out rollover time from comparison w/ the computer clock
1612 /// \attention In some cases this could fail to correctly calculate
1613 /// the rollover due to mismatches in the clocks... I think. This will
1614 /// be extremely rare but is something to keep in the back of your mind.
1615 Double_t dragon::LiveTimeCalculator::CalculateRuntime(midas::Database* db,
1616  const char* which,
1617  Double_t& start,
1618  Double_t& stop)
1619 {
1620  Int_t time0, time1, tclock;
1621  db->ReadValue("/Runinfo/Start time binary", time0); // computer clock start
1622  db->ReadValue("/Runinfo/Stop time binary", time1); // computer clock stop
1623  tclock = time1 - time0;
1624 
1625  // Read TSC start, stop times from the ODB
1626  double trigStart[2], trigStop[2];
1627  db->ReadArray("/Experiment/Run Parameters/TSC_TriggerStart", trigStart, 2);
1628  db->ReadArray("/Experiment/Run Parameters/TSC_TriggerStop", trigStop, 2);
1629 
1630  // Append rollovers to stop time
1631  const Double_t rolltime = 0xffffffff / 20e6; // 32-bit clock rollover in seconds
1632  int nroll = tclock / rolltime;
1633  for(int i=0; i<2; ++i) trigStop[i] += nroll*rolltime;
1634 
1635  // now calculte the appropriate run time
1636  switch(get_which_indx(which)) {
1637  case 0: // "head"
1638  stop = trigStop[0];
1639  start = trigStart[0];
1640  break;
1641  case 1: // "tail"
1642  stop = trigStop[1];
1643  start = trigStart[1];
1644  break;
1645  case 2: // "coinc"
1646  stop = *std::min_element(trigStop, trigStop + 2);
1647  start = *std::max_element(trigStart, trigStart + 2);
1648  break;
1649  default:
1650  utils::Error("CalculateRuntime", __FILE__, __LINE__)
1651  << "Invalid \"which\" specification: \"" << which << "\", valid options "
1652  << "are \"head\", \"tail\", or \"coinc\"";
1653  stop = 0;
1654  start = 0;
1655  break;
1656  }
1657  return stop - start;
1658 }

1890 ////////////////////////////////////////////////////////////////////////////////
1891 /// Calculate total busy time for coincidence events
1892 ///
1893 /// Calcluation of the busy time is performed as follows:
1894 /// First, sort the collection of events by the trigger time. Then sum up all of
1895 /// the busy times, but ignoring any overlap. To accomplish this, use the
1896 /// `std::accumulate` algorithm, with a custom "accumulator" function which
1897 /// ignores overlap by storing the last "end time" of an event and removing
1898 /// any overlap between the begin time of the present event and the previous
1899 /// end time. The code below shows how this is accomplished.
1900 /// \code
1901 /// struct AccumulateBusy {
1902 /// struct Output_t {
1903 /// Double_t fEnd;
1904 /// Double_t fTotalBusy;
1905 /// };
1906 /// Output_t operator() (const Output_t& current,
1907 /// const dragon::CoincBusytime::Event& additional);
1908 /// };
1909 ///
1910 /// inline AccumulateBusy::Output_t AccumulateBusy::operator()
1911 /// (const Output_t& current, const dragon::CoincBusytime::Event& additional)
1912 /// {
1913 /// Output_t out = current;
1914 /// if( current.fEnd < additional.fTrigger ) { // no overlap
1915 /// out.fTotalBusy += additional.fBusy;
1916 /// out.fEnd = additional.End();
1917 /// }
1918 /// else if( current.fEnd < additional.End() ) { // incomplete overlap
1919 /// out.fTotalBusy += (additional.End() - current.fEnd);
1920 /// out.fEnd = additional.End();
1921 /// }
1922 /// else { // complete overlap
1923 /// ;
1924 /// }
1925 /// return out;
1926 /// }
1927 ///
1928 /// Double_t dragon::CoincBusytime::Calculate()
1929 /// {
1930 /// if(!fIsSorted) Sort();
1931 /// AccumulateBusy::Output_t accum = { 0., 0. };
1932 /// accum = std::accumulate(fEvents.begin(), fEvents.end(), accum, AccumulateBusy());
1933 /// return accum.fTotalBusy / 1e6;
1934 /// }
1935 /// \endcode
1936 ///
1937 /// \returns Total coincidence busy time in seconds.
1938 Double_t dragon::CoincBusytime::Calculate()
1939 {
1940  if(!fIsSorted) Sort();
1941  AccumulateBusy::Output_t accum = { 0., 0. };
1942  accum = std::accumulate(fEvents.begin(), fEvents.end(), accum, AccumulateBusy());
1943  return accum.fTotalBusy / 1e6;
1944 }

2019 /// `E/m = cmd1 * (qB/m)<sup>2</sup> - [1/(2*u*c<sup>2</sup>)] * (E/m)<sup>2</sup>`

2022 /// \param md1Err Absolute error on MD1 field
2023 /// \param q Beam charge state

2042 }
