35 bool vme::Io32::unpack(const midas::Event& event, const char* bankName, bool reportMissing)
36 {
37  /*! Here is the portion of the MIDAS frontent where values are written to the "main" bank:
38  * \code
39  * *pdata32++ = 0xaaaa0020; // 0 - header and version
40  * *pdata32++ = trig_count-1; // 1 - event number, counting from 0
41  * *pdata32++ = trig_time; // 2 - trigger timestamp
42  * *pdata32++ = start_time; // 3 - readout start time
43  * *pdata32++ = end_time; // 4 - readout end time
44  * *pdata32++ = start_time-trig_time; // 5 - trigger latency (trigger time - start of readout time)
45  * *pdata32++ = end_time-start_time; // 6 - readout elapsed time
46  * *pdata32++ = end_time-trig_time; // 7 - busy elapsed time
47  * *pdata32++ = trigger_latch // 8 - dragon trigger latch
48  * \endcode
49  *
50  * The TSC4 bank is already unpacked in a midas::Event, so we can just copy the data over
51  * \param [in] event The midas event to unpack
52  * \param [in] bankName Name of the "main" IO32 bank
53  * \returns True if the event was successfully unpacked, false otherwise
54  */
55  int bank_len;
56  const int expected_bank_len = 9;
57  uint32_t* pdata32 =
58  event.GetBankPointer<uint32_t>(bankName, &bank_len, reportMissing, true);
59 
60  if (!pdata32) return false;
61 
62  if (bank_len != expected_bank_len) {
63  dutils::Error("vme::Io32::unpack", __FILE__, __LINE__) <<
64  "Bank length: " << bank_len << " != 8, skipping..." << DRAGON_ERR_FILE_LINE;
65  return false;
66  }
67 
68  uint32_t* data_fields[expected_bank_len]
69  = { &header, &trig_count, &tstamp, &start, &end,
70  &latency, &read_time, &busy_time, &trigger_latch };
71 
72  for (int i=0; i< bank_len; ++i) {
73  *(data_fields[i]) = *pdata32++;
74  }
75 
76  // Unpck TSC4 from midas::Event storage
77  tsc4.trig_time = event.TriggerTime();
78  event.CopyFifo(tsc4.fifo);
79  for(int j=0; j< 4; ++j)
80  tsc4.n_fifo[j] = tsc4.fifo[j].size();
81 
82  return true;

185 bool vme::V1190::unpack_data_buffer(const uint32_t* const pbuffer)
186 {
187  /*!
188  * \param [in] pbuffer Pointer to the data buffer word.
189  *
190  * A data buffer encodes the measurement value (i.e. the pulse time)
191  * for a single TDC measurement. The TDC is multi-hit, so more than one measurement
192  * per channel can be read in a single event.
193  *
194  * See below for bitpacking instructions.
195  */
196 
197  type = (*pbuffer >> 26) & READ1; /// - Bit 26 tells the measurement type (leading or trailing)
198  int ch = (*pbuffer >> 19) & READ7; /// - Bits 19-25 tell the channel number
199  if (ch >= MAX_CHANNELS) {
200  dutils::Error("vme::V1190::unpack_data_buffer", __FILE__, __LINE__)
201  << DRAGON_ERR_FILE_LINE << "Read a channel number (" << ch
202  << ") which is >= the maximum (" << MAX_CHANNELS << "). Skipping...\n";
203  return false;
204  }
205  if ( !(type==1 || type == 0) ) {
206  dutils::Error("vme::V1190::unpack_data_buffer", __FILE__, __LINE__)
207  << "\"type\" bitfield == " << type << ": Should be impossible, skipping event..."
208  << DRAGON_ERR_FILE_LINE;
209  return false;
210  }
211 
212  int32_t measurement = (*pbuffer >> 0) & READ19; /// - Bits 0 - 18 encode the measurement value
213 
214  if (type == 0) // leading edge
215  {
216  channel[ch].fLeading.push_back(measurement);
217  }
218  else // trailing edge
219  channel[ch].fTrailing.push_back(measurement);
220 
221  channel[ch].nleading = channel[ch].fLeading.size();
222  channel[ch].ntrailing = channel[ch].fTrailing.size();
223 
224  if(type == 0) fifo0.push_back(measurement, ch, channel[ch].nleading);
225  else fifo1.push_back(measurement, ch, channel[ch].ntrailing);
226 
227  return true;
228 }

260  "Hit lost in group 0 from L1 buffer overflow",

262  "Hit lost in group 1 from read-out FIFO overflow.",
263  "Hit lost in group 1 from L1 buffer overflow",
264  "Hit error have been detected in group 1.",
265  "Hit data lost in group 2 from read-out FIFO overflow.",
266  "Hit lost in group 2 from L1 buffer overflow",
267  "Hit error have been detected in group 2.",
268  "Hit lost in group 3 from read-out FIFO overflow.",
269  "Hit lost in group 3 from L1 buffer overflow",

293 bool vme::V1190::unpack_buffer(const uint32_t* const pbuffer, const char* bankName)
294 {
295  /*!
296  * \param [in] pbuffer Pointer to the buffer data
297  * \param [in] bankName Name of the midas bank being unpacked
298  * \returns True if the buffer was successfully unpacked, false otherwise
299  *
300  * V1190 buffers are 32 bit words. Bits 27-31 specify the type of data contained
301  * in the buffer. In this function, we read the buffer type and then handle appropriately.
302  */
303  bool success = true;
304  uint32_t type = (*pbuffer >> 27) & READ5;
305 
306  switch (type) {
307  case GLOBAL_HEADER: /// case GLOBAL_HEADER: read event counter from bits 5 - 26
308  count = (*pbuffer >> 5) & READ22;
309  break;
310  case GLOBAL_TRAILER: /// case GLOBAL_TRAILER: read status word from bits 24-26 and word count from bits 5-21
311  status = (*pbuffer >> 24) & READ3;
312  trailer_word_count = (*pbuffer >> 5) & READ16;
313  break;
314  case EXTENDED_TRIGGER_TIME: /// case EXTENDED_TRIGGER_TIME: read extended trigger from bits 0 - 26
315  extended_trigger = (*pbuffer >> 0) & READ27;
316  break;
317  case TDC_HEADER: /// case TDC_HEADER: read bunch id from bits 0 - 12, event id from bits 12 - 23
318  bunch_id = (*pbuffer >> 0) & READ12;
319  event_id = (*pbuffer >> 12) & READ12;
320  break;
321  case TDC_MEASUREMENT: /// case TDC_MEASUREMENT: See unpack_data_buffer()
322  success = unpack_data_buffer(pbuffer);
323  break;
324  case TDC_ERROR: /// case TDC_ERROR: See handle_error_buffer()
325  handle_error_buffer(pbuffer, bankName);
326  success = false;
327  break;
328  case TDC_TRAILER: /// case TDC_TRAILER: See unpack_footer_buffer()
329  unpack_footer_buffer(pbuffer, bankName);
330  break;
331  default: /// Bail out if we read an unknown buffer code
332  dutils::Error("vme::V1190::unpack_buffer", __FILE__, __LINE__)
333  << DRAGON_ERR_FILE_LINE << "Bank name: \"" << bankName
334  << "\": Unknown TDC buffer code: 0x" << std::hex << type << ". Skipping...\n";
335  success = false;
336  break;
337  }
338  return success;
339 }
