29 //////////////////////// Class dragon::Kin2Body ////////////////////////////////
30 /// Class for relativistic 2 body reaction kinematics
31 /*!
32  * Calculates relativistic kinematic relationships for 2 body reactions.
33  * Consider the reaction
34  * \f{equation}{
35  * m_2(m_1,m_3)m_4
36  * \f}
37  * where, by convention, the projectile with mass \f$m_1\f$ and kinetic energy
38  * \f$T_b\f$ and is incident on a (stationary) target of mass \f$m_2\f$. The
39  * kinematics of this reaction can be described by its (invariant) squared
40  * 4-momentum
41  * \f{equation}{
42  * S = \left(\sum_i E_i\right)^2 - \left(\sum_i \vec{p}_{i}\right)^2
43  * \f}
44  */

85 void dragon::Kin2Body::Init(const char* projectile, const char* target,
86  Double_t energy, const char* frame, Int_t qb)
87 {
88  TAtomicMassTable mt; // AME16
89  fProj = mt.GetNucleus(projectile);
90  fTgt = mt.GetNucleus(target);
91  fRec = mt.GetNucleus(fProj->fZ+fTgt->fZ, fProj->fA+fTgt->fA);
92  fRxnString = Form("{}^{%i}%s(^{%i}%s,#gamma)^{%i}%s",fTgt->fA,fTgt->fSymbol,fProj->fA,fProj->fSymbol, fRec->fA, fRec->fSymbol);

95  fEjString = "#gamma";
96  fRecString = Form("{}^{%i}%s",fRec->fA,fRec->fSymbol);
97  fM1 = mt.IonMass(fProj->fZ, fProj->fA, 0) / 1.e3;
98  fM1amu = mt.IonMassAMU(fProj->fZ, fProj->fA, 0);
99  fM2 = mt.IonMass(fTgt->fZ, fTgt->fA, 0) / 1.e3;
100  fM3 = 0;
101  fM4 = mt.IonMass(fRec->fZ, fRec->fA, 0) / 1.e3;
102  fqb = qb;
103  fQrxn = mt.QValue(projectile, target, "g", false) / 1.e3;
104  Set4Mom(energy, frame);
105  fPcm = sqrt( (pow(fS - fM1*fM1 - fM2*fM2, 2) - 4*pow(fM1*fM2, 2) ) / (4*fS) );
106  fPprime = sqrt( (pow(fS - fM4*fM4, 2) ) / (4*fS) );
107  fChi = log( (fPcm + sqrt(fM2*fM2 + fPcm*fPcm) ) / fM2 );
108  fEcm = sqrt(fS) - fM1 - fM2;
109  fEx = fEcm + fQrxn;
110  fTb = ( fS - pow(fM1 + fM2,2) ) / (2*fM2);
111  fPb = sqrt(pow(fTb,2) + 2*fTb*fM1);
112  fTtgt = ( fS - pow(fM1 + fM2,2) ) / (2*fM1);
113  fTbA = fTb / fProj->fA;
114  fV2b = fTb / fM1amu;
115  if (fqb != 0){
116  fBrho = ( 3.3356 / (1.e3*fqb) )*sqrt( ( (fS - pow(fM1+fM2,2)) / (2*fM2) )*( (fS - pow(fM1+fM2,2)) / (2*fM2) + 2*fM1) );
117  } else {
118  fBrho = 0;
119  }
120  fEb = fTb + fM1;
121  fGamma_b = fEb / fM1;
122  fBeta_b = sqrt(1 - 1/pow(fGamma_b,2));
123  fMcpTofb = dragon::Constants::Lmcp() / (fBeta_b*TMath::C())*1.0e9;
124  fSepTofb = dragon::Constants::Ldra() / (fBeta_b*TMath::C())*1.0e6;
125  fE90rec = sqrt(fPb*fPb + fM4*fM4);
126  fGamma90rec = fE90rec / fM4;
127  fBeta90rec = sqrt(1 - 1/pow(fGamma90rec,2));
128  fMcpTofrec90 = dragon::Constants::Lmcp() / (fBeta90rec*TMath::C())*1.0e9;
129  fSepTofrec90 = dragon::Constants::Ldra() / (fBeta90rec*TMath::C())*1.0e6;
130  fE0rec = sqrt( pow(fPb*(1 - fEx / sqrt(fEb*fEb - fM1*fM1)),2) + fM4*fM4 );
131  fGamma0rec = fE0rec / fM4;
132  fBeta0rec = sqrt(1 - 1/pow(fGamma0rec,2));
133  fMcpTofrec0 = dragon::Constants::Lmcp() / (fBeta0rec*TMath::C())*1.0e9;
134  fSepTofrec0 = dragon::Constants::Ldra() / (fBeta0rec*TMath::C())*1.0e6;
135  fE180rec = sqrt( pow(fPb*(1 + fEx / sqrt(fEb*fEb - fM1*fM1)),2) + fM4*fM4 );
136  fGamma180rec = fE180rec / fM4;
137  fBeta180rec = sqrt(1 - 1/pow(fGamma180rec,2));
138  fMcpTofrec180 = dragon::Constants::Lmcp() / (fBeta180rec*TMath::C())*1.0e9;
139  fSepTofrec180 = dragon::Constants::Ldra() / (fBeta180rec*TMath::C())*1.0e6;

197 Double_t dragon::Kin2Body::CalcTLabTheta(Double_t theta, const char* which, Bool_t negative)
198 {
199  Double_t mass;
200  if (strncmp(which, "recoil", 6) == 0){
201  mass = fM4;
202  } else {
203  mass = fM3;
204  }
205 
206  if (theta == 90.0){
207  if (mass*sinh(fChi) / fPprime > 1){
208  return 0.0;
209  } else {
210  Double_t p = fPprime*sqrt(1 - pow(mass*sinh(fChi) / fPprime,2) / (1 + pow(sinh(fChi),2)) );
211  Double_t T = sqrt(p*p + mass*mass) - mass;
212  if (T < 0.001) {
213  return 0;
214  } else {
215  return T;
216  }
217  }
218  }
219 
220  if (theta >= GetMaxAngle(which)) {
221  theta = GetMaxAngle(which);
222  theta /= 180.;
223  theta *= TMath::Pi();
224  Double_t pe = cos(theta)*sinh(fChi)*sqrt(mass*mass + fPprime*fPprime);
225  pe /= ( 1+pow(sin(theta)*sinh(fChi),2) );
226  return sqrt(pe*pe + mass*mass) - mass;
227  } else if (!negative){
228  theta /= 180.;
229  theta *= TMath::Pi();
230  Double_t pe = cos(theta)*sinh(fChi)*sqrt(mass*mass + fPprime*fPprime);
231  pe += cosh(fChi)*sqrt( fPprime*fPprime - pow(mass*sin(theta)*sinh(fChi),2) );
232  pe /= ( 1+pow(sin(theta)*sinh(fChi),2) );
233  return sqrt(pe*pe + mass*mass) - mass;
234  } else {
235  theta /= 180.;
236  theta *= TMath::Pi();
237  Double_t pe = cos(theta)*sinh(fChi)*sqrt(mass*mass + fPprime*fPprime);
238  pe -= cosh(fChi)*sqrt( fPprime*fPprime - pow(mass*sin(theta)*sinh(fChi),2) );
239  pe /= ( 1+pow(sin(theta)*sinh(fChi),2) );
240  return sqrt(pe*pe + mass*mass) - mass;
241  }
