21 Bool_t dragon::ASelector::Process(Long64_t)
22 {
23  //! Event-by-event actions.
24  ///
25  /// The Process() function is called for each entry in the tree (or possibly
26  /// keyed object in the case of PROOF) to be processed. The entry argument
27  /// specifies which entry in the currently loaded tree is to be processed.
28  /// It can be passed to either dragon::HeadSelector::GetEntry() or TBranch::GetEntry()
29  /// to read either all or the required parts of the data. When processing
30  /// keyed objects with PROOF, the object is already loaded and is available
31  /// via the fObject pointer.
32  ///
33  /// This function should contain the "body" of the analysis. It can contain
34  /// simple or elaborate selection criteria, run algorithms on the data
35  /// of the event and typically fill histograms.
36  ///
37  /// The processing can be stopped by calling Abort().
38  ///
39  /// Use fStatus to set the return value of TTree::Process().
40  ///
41  /// The return value is currently not used.
42  ///
43  /// \note Derived classes should always implement this method.
44  ///
45  /// \warning When a selector is used with a TChain:
46  /// in the Process, ProcessCut, ProcessFill function, you must use
47  /// the pointer to the current Tree to call `GetEntry(entry).`
48  /// entry is always the local entry number in the current tree.
49  /// Assuming that fChain is the pointer to the TChain being processed,
50  /// use `fChain->GetTree()->GetEntry(entry);`
51  ///
52  AbstractMethod("Process");
53  return kTRUE;
54 }
