65  /// Calculate weighted average of measurements
66  template <class InputIterator>
67  UDouble_t MeasurementWeightedAverage(InputIterator begin, InputIterator end)
68  {
69  ///
70  /// Procedure:
71  /// -# Calculate `num` = `\Sum_i { x_i * (1./sigma_i^2) }
72  /// -# Calculate `den` = `\Sum_i { 1./sigma_i^2 }
73  /// -# Final result: nominal value = `num/den`, error = `sqrt(den)`
74  ///
75  /// \attention Assumes symmetric errors.
76  ///
77  Double_t num = 0, den = 0;
78  while (begin != end) {
79  Double_t val = begin->GetNominal(), weight = 1. / pow(begin->GetRelErrLow(), 2);
80  num += val*weight;
81  den += weight;
82  ++begin;
83  }
84  return UDouble_t (num/den, sqrt(1./den));
85  }

582  /// Class to calculate total busy time for coincidence events
583  /*!
584  * For coincidence events, the total busy time is the sum or times during which the head _or_
585  * tail DAQ is busy. Any genuine coincidence triggers arriving when either the head or tail is
586  * busy will not be marked as such. This covers all possible classes of lost coincidence events:
587  * - Coincidences arriving when both the head and tail are busy: the trigger is simply not recorded.
588  * - Coincidences arriving when either the head or is tail is busy but not the other: the coincidence
589  * will be incorrectly marked as a singles event.
590  * - Coincidences whose match window is interrupted by an uncorrelated singles event. For example, a
591  * capture reaction happens, triggers the BGOs and then uncorrelated leaky beam triggers the end
592  * detector before the recoil gets there. This will still be marked as a coincidence event in the
593  * analysis, but the head event will not be correlated with the correct tail one. Thus the _real_
594  * coincidence has been lost, albeit to a false (random) coincidence.
595  *
596  * The coincidence dead time is then the sum the time during which the head or tail is busy divided by
597  * the "coincidence runtime", i.e. the time during which both the head and tail are accepting triggers.
598  *
599  * For information about how the busy time is calculated in practice, see the documentation on
