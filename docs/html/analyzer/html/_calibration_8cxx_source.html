150 /// Helper routine to find triple alpha peaks in a spectrum
151 std::vector<Double_t> dutils::DsssdCalibrator::FindPeaks(TH1* hst,
152  Double_t sigma,
153  Double_t threshold) const
154 {
155  std::vector<Double_t> peaks;
156  TSpectrum spectrum;
157  spectrum.Search(hst, sigma, "goff", threshold);
158  Int_t npeaks = spectrum.GetNPeaks();
159  std::vector<int> indx(npeaks);
160  TMath::Sort(npeaks, spectrum.GetPositionX(), &indx[0], kFALSE);
161  for(Int_t i = 0; i < npeaks; ++i) {
162  peaks.push_back(*(spectrum.GetPositionX() + indx[i]));
163  }
164  return peaks;
165 }

167 ////////////////////////////////////////////////////////////////////////////////
168 /// Helper routine to fit alpha energy vs. ADC channel
169 /// \param pklow Low edge (in uncalibrated channel number) of the region
170 /// where the peak finding algorithm should search for triple alpha peaks
171 /// \param pkhigh High edge (in uncalibrated channel number) of the region
172 /// where the peak finding algorithm should search for triple alpha peaks
173 /// \param Sigma Minimum width of peaks to be found by the searching algorithm
174 /// \param threshold Minimum peak height for the peak searching algorithm
175 /// \note See `TSpectrum::Search` in ROOT's class documentation for more info on
176 /// the _sigma_ and _threshold_ parameters
177 Int_t dutils::DsssdCalibrator::Run(Int_t nbins, Double_t pklow, Double_t pkhigh,
178  Double_t sigma, Double_t threshold, Bool_t grid)
179 {
180  if(!fTree) return 0;
181  TH1F hpeaks("hpeaks", "", nbins, pklow, pkhigh);
182  Int_t retval = 0;
183  for(Int_t i = 0; i < NDSSSD; ++i) {
184  char buf[16];
185  // Double_t slope = GetOldParams(i).slope, offset = GetOldParams(i).offset;
186  // sprintf(buf, "(dsssd.ecal[%d] - %f) / %f", i, offset, slope);
187  sprintf(buf, "dsssd.ecal[%d]", i);
188  fTree->Project("hpeaks", buf, "", "goff");
189  std::vector<Double_t> peaks = FindPeaks(&hpeaks, sigma, threshold);
190  if(peaks.size() != 3) {
191  std::cerr << "Number of peaks found for channel " << i << ": " << peaks.size()
192  << " != 3, skipping!\n";
193  fParams[i].slope = -1.99999;
194  fParams[i].offset = 0.0;
195  continue;
196  } else {
197  ++retval;
198  }
199  for(int j = 0; j < 3; ++j)
200  fPeaks[i][j] = peaks[j];
201 
202  FitPeaks(i, grid);
203  }
204  GainMatch();
205  return retval;
206 }

210 void dutils::DsssdCalibrator::GainMatch()
211 {
212  fMaxSlope = 0;
213  for (Int_t i = 0; i < NDSSSD; ++i){

215  fMaxSlope = fParams[i].slope;
216  fMinChan = i;

219  }
220  }
221  for(Int_t i = 0; i < NDSSSD; i++){
222  if ( i == fMinChan){
223  fParams[i].slope = 1.0;
224  }
225  else if ( fParams[i].slope < 0 ) {
226  continue;
227  }
228  else{
229  fParams[i].slope /= fMaxSlope;
230  }
231  fParams[i].offset /= fMaxSlope;
232  }

235 ////////////////////////////////////////////////////////////////////////////////
236 /// Run the full calibration
237 /// \param ch DSSSD channel
238 /// \param grid boolean specifying whether detector is Tengblad or standard
239 /// design
240 void dutils::DsssdCalibrator::FitPeaks(Int_t ch, Bool_t grid)
241 {
242  if((UInt_t)ch > NDSSSD - 1) return;
243  const Double_t alphaEnergies[3] = {5.15659,5.48556,5.80477}; /* primary alpha energies of triple alpha source in MeV. */
244  const Double_t dLayer = 374.0e-5; /* dead layer thickness (Al equivalent) in mm (from C. Wrede's thesis). */
245  const Double_t dLayer_grid = 5.0e-5; /* dead layer thickness in mm (Si) of Tengblad design (girdded) DSSSD */
246  const Double_t dEdx_Al[3] = {160.1167,154.1106,148.9053}; // stopping powers (MeV/mm) of alphas in Al corresponding to above energies according to SRIM 2008
247  const Double_t dEdx_Si[3] = {139.006,133.701,129.296}; // stopping powers (MeV/mm) of alphas in Si corresponding to above energies according to SRIM 2008
248  // const Double_t aEloss[3] = {138.080,133.060,128.546}; /* Stopping Powers in MeV/mm. */
249  Double_t aEnergy[3]; /* Actual energy deposited */
250  if (grid) {
251  for (int i = 0; i < 3; i++)
252  aEnergy[i] = alphaEnergies[i] - dEdx_Si[i]*dLayer_grid;
253  }
254  else {
255  for (int i = 0; i < 3; i++)
256  aEnergy[i] = alphaEnergies[i] - dEdx_Al[i]*dLayer;
257  }
258  TGraph gr(3);
259  for(Int_t i = 0; i < gr.GetN(); ++i) {
260  gr.SetPoint(i, GetPeak(ch, i), aEnergy[i]);
261  }
262  TFitResultPtr result = gr.Fit("pol1", "qns");
263  fParams[ch].slope = result->Value(1);
264  fParams[ch].offset = result->Value(0);
265 }
