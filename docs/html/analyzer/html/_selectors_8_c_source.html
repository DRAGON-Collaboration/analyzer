56 /// Example selector class for scaler events
57 /**
58  * Fills a histogram with the SB 0 counts, then calculates the
59  * scaler sum in two different ways.
60  */
61 class ExampleScalerSelector : public dragon::ScalerSelector
62 {
63 public:
64  /// 1D histogram of SB 0 rate
65  TH1* fSb0;
66 
67 public:
68  /// Initialize fSb0 pointer to NULL
69  ExampleHeadSelector(): fSb0(0) { }
70 
71  /// Called at the beginning of a loop
72  void Begin(TTree*)
73  {
74  /// Set fSb0 to a new histogram
75  if(fSb0) delete fSb0;
76  fSb0 = new TH1I("sb0_count", "SB 0 counts per readout", 3600, 0, 3600);
77  }
78 
79  /// Called event-by-event in a loop
80  Bool_t Process(Long64_t entry)
81  {
82  ///
83  /// Reads event data, increments histogram bin
84 
85  if(!fChain) Abort("fChain == 0");
86  if(!fSb0) Abort("fSb0 == 0");
87 
88  this->GetEntry(entry); // reads all branches
89 
90  /// \note Two "gotchas" involved in filling the histogram
91  /// 1. The bin number should be `entry + 1`, not `entry`. This is
92  /// because bin 0 is for underflow counts. In other words, ROOT
93  /// histograms start counting from 1.
94  /// 2. We set the bin content to `count[10]` and not `rate[10]`. This is
95  /// so that the integral of the histogram will match the actual sum
96  /// of counts. Had we used `rate` instead, we would have to correct the
97  /// integral for the difference bewteen the frontend CPU clock (which
98  /// triggers a scaler read every second) and the IO32 clock (which calculates
99  /// a more precise time in between scaler reads and uses this to calculate
100  /// the `rate` parameter).
101  ///
102  fSb0->SetBinContent(entry + 1, count[10]); // [10] == SB 0
103 
104  return kTRUE; // (ignored)
105  }
106 
107  /// Called at the end of processing
108  void Terminate()
109  {
110  /// Draw the SB 0 count histogram, calculate the sum both from it's
111  /// integral and from the last entry of `sum[10]`.
112  fSb0->Draw();
113 
114  fChain->GetEntry(fChain->GetEntries() - 1);
115  std::cout << "\nSB 0 sum:\nMethod 1 (integral of 'count[10]'): "
116  << fSb0->Integral() << "\nMethod 2 (`sum[10] for last event): "
117  << sum[10] << std::endl;
118  }
119 
120  /// Deltes fSb0
121  ~ExampleHeadSelector()
122  {
123  if(fSb0) delete fSb0;
124  }
125 };
