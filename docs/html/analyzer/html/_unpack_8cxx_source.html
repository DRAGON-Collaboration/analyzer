143 std::vector<int32_t> dragon::Unpacker::UnpackMidasEvent(void* header, char* data)
144 {
145  fUnpacked.clear();
146  midas::Event::Header* evtHeader = reinterpret_cast<midas::Event::Header*>(header);
147  switch (evtHeader->fEventId)
148  {
149  case DRAGON_HEAD_EVENT:
150  {
151  if(IsSinglesMode()) {
152  midas::Event event(header, data, evtHeader->fDataSize);
153  UnpackHead(event);
154  }
155  else {
156  midas::Event event(header, data, evtHeader->fDataSize, fHead->variables.bk_tsc, GetCoincWindow());
157  fQueue->Push(event, fDiag);
158  }
159  break;
160  }
161  case DRAGON_TAIL_EVENT:
162  {
163  if(IsSinglesMode()) {
164  midas::Event event(header, data, evtHeader->fDataSize);
165  UnpackTail(event);
166  }
167  else {
168  midas::Event event(header, data, evtHeader->fDataSize, fTail->variables.bk_tsc, GetCoincWindow());
169  fQueue->Push(event, fDiag);
170  }
171  break;
172  }
173  case DRAGON_HEAD_SCALER:
174  {
175  midas::Event event(header, data, evtHeader->fDataSize);
176  UnpackHeadScaler(event);
177  break;
178  }
179  case DRAGON_TAIL_SCALER:
180  {
181  midas::Event event(header, data, evtHeader->fDataSize);
182  UnpackTailScaler(event);
183  break;
184  }
185  case DRAGON_AUX_SCALER:
186  {
187  midas::Event event(header, data, evtHeader->fDataSize);
188  UnpackAuxScaler(event);
189  break;
190  }
191  case DRAGON_EPICS_EVENT:
192  {
193  midas::Event event(header, data, evtHeader->fDataSize);
194  UnpackEpics(event);
195  break;
196  }
197  case DRAGON_EPICS_SCALER:
198  {
199  // EPICS virtaul scalers data are redundant, so do nothing
200  break;
201  }
202  case MIDAS_BOR:
203  {
204  midas::Database db(data, evtHeader->fDataSize);
205  UnpackRunParameters(db);
206  break;
207  }
208  case MIDAS_EOR:
209  {
210  midas::Database db(data, evtHeader->fDataSize);
211  UnpackRunParameters(db);
212  break;
213  }
214  default:
215  {
216  utils::Warning("UnpackBuffer", __FILE__, __LINE__)
217  << "Unkonwn event ID: " << evtHeader->fEventId;
218  break;
219  }
220  }
221  /// \returns The result of GetUnpackedCodes() after this event
222  return fUnpacked;
223 }
