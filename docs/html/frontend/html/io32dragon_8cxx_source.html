179 void Io32dragon::read_tsc4 (char *pevent)
180 {
186  uint32_t* pdata32;
187  bk_create(pevent, fTscBankName, TID_DWORD, (void**)&pdata32);
188 
189  uint32_t tscctl;
190  *pdata32++ = io32->read32(IO32_REVISION); // firmware revision
191  *pdata32++ = io32->read32(IO32_TS); // 20MHz timestamp
192  *pdata32++ = io32->read32(IO32_TSC4_ROUTE); // TSC routing
193  *pdata32++ = tscctl = io32->read32(IO32_TSC4_CTRL); // TSC control
194 
195  int wc = tscctl & 0x7fff; // Number of words: bits 14..0
196  int overflow = tscctl & 0x8000; // Overflow flag: bit 15
197  uint32_t tsch = (tscctl >> 16) & 0xff; // Bits 23..16: bits 35..28 of TSC
198  *pdata32++ = gRollh.check(tsch); // Rollover counter for upper bits
199 
200  for (int i=0; i<wc; i++) {
201  *pdata32++ = io32->read32(IO32_TSC4_FIFO); // read TSC FIFO (ch & lower 30 counter bits)
202  if(0) printf("fifo %d %u\n", i, (*(pdata32-1) & 0x3fffffff));
203  }
204 
205  if (overflow) {
206  io32->write32(IO32_TSC4_FIFO, 0); // clear TSC overflow flag
207  *pdata32++ = 0xFFFFFFFF; // write a TSC overflow marker
208  if(1) cm_msg(MERROR, frontend_name, "IO32 TSC4 timestamp fifo overflow, tsc status 0x%08x", tscctl);
209  }
210 
211  bk_close(pevent, pdata32);
212 }

226  switch (fType)
227  {
228  case kMaster:

230  io32->SetNimOutputFunction(1, 1); // set NIM output 1 function 1 - daq busy
231  io32->SetNimOutputFunction(2, 0); // set NIM output 2 function 0 - nimout[2]
232  io32->SetNimOutputFunction(3, 0); // set NIM output 3 function 0 - 40 MHz clock
233  break;
234  case kSlave:

236  io32->SetNimOutputFunction(1, 1); // set NIM output 1 function 1 - daq busy
237  io32->SetNimOutputFunction(2, 0); // set NIM output 2 function 0 - nimout[2]
238  io32->SetNimOutputFunction(3, 1); // set NIM output 3 function 1 - nimout[3]
239  break;
240  case kScaler: // (duplicate slave)
241  io32->SetNimOutputFunction(0, 1); // set NIM output 0 function 1 - 20 MHz clock
242  io32->SetNimOutputFunction(1, 1); // set NIM output 1 function 1 - daq busy
243  io32->SetNimOutputFunction(2, 0); // set NIM output 2 function 0 - nimout[2]
244  io32->SetNimOutputFunction(3, 1); // set NIM output 3 function 1 - nimout[3]
245  break;
246  default:
247  assert (fType == kMaster || fType == kSlave || fType == kScaler);
248  break;
249  }
250 
251  io32->write32(IO32_NIMIN, 0xFFFF); // clear NIM input latch
252 
253  {
254  std::string path = fOdbVmeBase + std::string("/Settings/IO32/TSC_route");
255  INT tsc4route[4];
256  INT size = sizeof(tsc4route);
257  INT status = db_get_value(odbHandle, 0, path.c_str(), tsc4route, &size, TID_INT, FALSE);
258

261  return status;
262  }
263

265  {
266  case kMaster:
267  // Set up TSC4 routing: inputs = {from ODB}; clock input = 8 (n/a); clock option = 0
268  init_tsc4 (get_io32(), tsc4route[0], tsc4route[1], tsc4route[2], tsc4route[3], 8, 0);
269  break;
270  case kSlave:
271  // Set up TSC4 routing: inputs = {from ODB}; clock input = 8 (n/a); clock option = 0
272  init_tsc4 (get_io32(), tsc4route[0], tsc4route[1], tsc4route[2], tsc4route[3], 5, 3);
273  break;
274  case kScaler:
275  // (same as master)

277  break;
278  default:
279  assert (fType == kMaster || fType == kSlave || fType == kScaler);
280  break;
281  }
282  }

285 
286  // Reset scalers
287  io32->write32(IO32_COMMAND, IO32_CMD_RESET_TS);
288  io32->write32(IO32_COMMAND, IO32_CMD_RESET_SCALERS);

336 
337  uint32_t rev = io32->read32(IO32_REVISION); // read firmware revision
338  cm_msg(MINFO, frontend_name, "trigger: VMENIMIO32 firmware revision 0x%08x", rev);

340  int size = sizeof(int);

342  // program the pulse generator

345  {

358  if (pulser_enabled) {

360  io32->write32(IO32_PULSER, pulser_period);

367  {
368  std::string pathDelay = fOdbVmeBase + std::string("/Settings/IO32/Tdc_delay");

370 
371  db_get_value(odbHandle, 0, pathDelay.c_str(), &delay, &size, TID_INT, TRUE);
372  db_get_value(odbHandle, 0, pathWidth.c_str(), &width, &size, TID_INT, TRUE);
373  }
374 
375  delay /= 10; // convert to ns
376  width /= 10;
377  cm_msg(MINFO, frontend_name, "trigger: TDC trigger delay %d [ns], ADC gate width %d [ns]", delay*10, width*10);
378  io32->write32(4*51, (width<<16)|(delay));
379 
380  // set up trigger conditions
381  uint32_t w = 0;
382  w |= 0x80000000; // enable DRAGON trigger function
383  w |= 0x40000000; // enable DRAGON internal busy
384 
385  // read masked inputs from odb settings
386  uint32_t trigger_mask = 0xFF;
387  {
388  INT status = 0;
389  std::string path = fOdbVmeBase + std::string("/Settings/IO32/ChannelEnable");
390  trigger_mask = get_trigger_mask (odbHandle, path.c_str(), &status);
391  if(status != DB_SUCCESS) return status;
392  }
393 
394  w |= (trigger_mask & 0xFF) << 16;
395 
396  cm_msg(MINFO, frontend_name, "trigger: DRAGON Trigger control word 0x%08x", w);
397  io32->write32(4*58, w);
398 
399  gRollh.reset(); // reset upper tsc rollover counter
400  return SUCCESS;
401 }
